		&= \frac{1}{2} ||F(x)||^2_2 + \frac{c}{2} ||h(x) + \lambda/c ||^2_2 - \frac{1}{2c} ||\lambda||^2_2 \\
		&= \frac{c}{2} \Big|\Big|
		\begin{bmatrix}
			F(x)/\sqrt{c} \\
			h(x) + \lambda/c
		\end{bmatrix} \Big|\Big|^2
	\end{aligned}
\end{equation*}
